.data

   MATRIZ BYTE 5, 10, 15
          BYTE 7, 8, 9
          BYTE 3, 6, 11

  FILAS DWORD 3
  COLUMNAS DWORD 3
  N DWORD 0
  NUMEROS DB "EL NUMERO SE REPITE UN TOTAL DE ", 0
  INGRESA_N DB  "INGRESA UN NUMERO", 0
  EFILA DB " DE LA FILA ", 0
  RES DB " VECES ", 0
; Área de Declaración de Variables
.code
     main25 PROC
          ;lógica del Programa
            ;POR FILA, ENCONTRAR EL VALOR MAYOR Y EL MENOR

          MOV EDX, OFFSET INGRESA_N
          CALL WRITESTRING
          CALL CRLF
          CALL READINT
          MOV N, EAX
          CALL CRLF
          MOV ECX, 0
          MOV EBX, 0 ;SERA EL CONTADOR DE CUANTAS VECES SE REPITE
          CICLO:
               MOV EAX, ECX
               MUL FILAS
               MOV EDI, EAX
               PUSH ECX
               MOV ECX, 0
               CICLO2:
                    MOV ESI, ECX
                    MOV EAX, 0
                    MOV AL, MATRIZ[ESI + EDI]
                    CMP EAX, N
                    JGE REPETICION
                    CONTINUAR:
                    INC ECX
                    CMP ECX, COLUMNAS
                    JL CICLO2
                    ;SALE DEL CICLO 2
                    POP ECX
                    INC ECX
                    CMP ECX, FILAS
                    JL CICLO
                    JMP RESULTADO

                    REPETICION:
                    ADD EBX, 1
                    JMP CONTINUAR
 
                  RESULTADO:
                    MOV EDX, OFFSET NUMEROS
                    CALL WRITESTRING
                    MOV EAX, 0
                    MOV EAX, EBX
                    CALL WRITEDEC
                    MOV EDX, OFFSET RES
                    CALL WRITESTRING
                    CALL CRLF
 
          exit
     main25 ENDP
     END main25
