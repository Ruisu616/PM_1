INCLUDE Irvine32.inc
;INTEGRANTES DEL EQUIPO
;Gonzalez Saldivar Luis Roberto
.data

   MATRIZ BYTE 5, 10, 15
          BYTE 7, 8, 9
          BYTE 3, 6, 11

  FILAS DWORD 3
  COLUMNAS DWORD 3
  MENOR DWORD 0
  MAYOR DWORD  0
  EMAYOR DB "EL NUMERO MAYOR DEL ARREGLO ES:", 0
  EMENOR DB " EL NUMERO MENOR DEL ARREGLO ES:", 0
  ECOLUMNA DB " DE LA COLUMNA ", 0
  RES DB " ES: ", 0
; Área de Declaración de Variables
.code
     main24 PROC
          ;lógica del Programa
            ;POR COLUMNAS, ENCONTRAR EL VALOR MAYOR Y EL MENOR
          
          MOV ECX, 0
          MOV ESI, ECX
          MOV EDI, ECX
          MOV ECX, 0
          CICLO:
               MOV EAX, ECX
               MOV ESI, EAX
               PUSH ECX
               MOV ECX, 0
               MOV EAX, 0
               MOV AL, MATRIZ[ESI + EDI]
               MOV MAYOR, EAX
               MOV MENOR, EAX
               CICLO2:
                    MOV EAX, ECX
                    MUL FILAS
                    MOV EDI, EAX
                    MOV EAX, 0
                    MOV AL, MATRIZ[ESI + EDI]
                    CMP EAX, MAYOR
                    JG MAYOR_11
                    CMP EAX, MENOR
                    JL MENOR_11
                    CONTINUAR2:
                    INC ECX
                    CMP ECX, FILAS
                    JL CICLO2
                    ;SALE DEL CICLO 2
                    MOV EDX, OFFSET EMAYOR
                    CALL WRITESTRING
                    MOV EDX, OFFSET ECOLUMNA
                    CALL WRITESTRING
                    POP EAX
                    INC EAX
                    CALL WRITEDEC
                    DEC EAX
                    PUSH EAX
                    MOV EDX, OFFSET RES
                    CALL WRITESTRING
                    MOV EAX, MAYOR
                    CALL WRITEDEC
                    CALL CRLF
                    MOV EDX, OFFSET EMENOR
                    CALL WRITESTRING
                    MOV EDX, OFFSET ECOLUMNA
                    CALL WRITESTRING
                    POP EAX
                    INC EAX
                    CALL WRITEDEC
                    DEC EAX
                    PUSH EAX
                    MOV EDX, OFFSET RES
                    CALL WRITESTRING
                    MOV EAX, MENOR
                    CALL WRITEDEC
                    CALL CRLF
                    POP ECX
                    INC ECX
                    CMP ECX, COLUMNAS
                    JL CICLO
                    JMP RESULTADO

                    MAYOR_11:
                    MOV MAYOR, EAX
                    JMP CONTINUAR2
                    MENOR_11:
                    MOV MENOR, EAX 
                    JMP CONTINUAR2
 
                  RESULTADO:
          exit
     main24 ENDP
     END main24
